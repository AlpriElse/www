public class Course {
  private String name;
  public Course(String setName) {
    this.name = setName;
  }
  public void printName() {
    System.out.println(this.name);
  }
}
public class Example {
  public static void main(String[] unused) {
    Course CS125 = new Course("CS 125");
    CS125.printName();
  }
}
