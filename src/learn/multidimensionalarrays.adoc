char[][] currentBoard =
{{'X', 'O', '.'}, {'O', 'X', '.'}, {'O', '.', 'X'}};
static void printBoard(char[][] board) {
  for (int i = 0; i < board.length; i++) {
    for (int j = 0; j < board[i].length; j++) {
      System.out.print(board[i][j]);
    }
    System.out.println();
  }
}
printBoard(currentBoard);
